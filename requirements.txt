import torch
from transformers import AutoModelForCausalLM, AutoTokenizer, TrainingArguments, Trainer
from datasets import Dataset

# Dataset with ladder logic examples
data = [
    {"text": "Generate ladder logic to turn on a motor when a start button is pressed and turn it off when a stop button is pressed.\nLadder Logic:\n|----[ ]----| |----( )----|\n|   START   | |    MOTOR  |\n|           | |           |\n|----[ ]----| |           |\n|   STOP    | |           |"},
    {"text": "Generate ladder logic to turn on a bulb when a button is pressed.\nLadder Logic:\n|----[ ]----( )----|\n|    BTN     BULB  |"},
    {"text": "Generate ladder logic to latch a bulb on when a button is pressed.\nLadder Logic:\n|----[ ]----| |----( )----|\n|    BTN    | |    BULB   |"},
]

# Convert data to Hugging Face Dataset
dataset = Dataset.from_list(data)

# Load pre-trained model and tokenizer
model_name = "deepseek-ai/deepseek-coder-1.3b-instruct"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(model_name)

# Tokenization function
def tokenize_function(examples):
    tokens = tokenizer(examples["text"], truncation=True, max_length=64, padding="max_length")
    tokens["labels"] = tokens["input_ids"].copy()
    return tokens

# Tokenize the dataset
tokenized_dataset = dataset.map(tokenize_function, batched=True)

# Training arguments
training_args = TrainingArguments(
    output_dir="./fast_finetune",
    per_device_train_batch_size=1,
    num_train_epochs=1,
    max_steps=10,
    logging_steps=1,
    use_cpu=True,  # Use this instead of no_cuda
    fp16=False
)

# Trainer
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=tokenized_dataset,
)

# Fine-tune the model
trainer.train()

# Save the fine-tuned model and tokenizer
trainer.save_model("./fast_finetune")
tokenizer.save_pretrained("./fast_finetune")

# Function to generate ladder logic
def generate_response(prompt):
    inputs = tokenizer(prompt, return_tensors="pt")
    outputs = model.generate(**inputs, max_length=64)
    return tokenizer.decode(outputs[0], skip_special_tokens=True)

# Generate ladder logic for a new prompt
prompt = "Generate ladder logic to turn on a laserlight when a start button is pressed."
generated_ladder_logic = generate_response(prompt)

# Save the generated ladder logic to a .txt file
with open("generated_ladder_logic_laser.txt", "w") as file:
    file.write(generated_ladder_logic)

print("Generated Ladder Logic saved to 'generated_ladder_logic_new.txt'")
